python
import os
import psycopg2
from fastapi import FastAPI, BackgroundTasks, HTTPException
from pydantic import BaseModel
from typing import Optional, Dict, Any
from neo4j import GraphDatabase

app = FastAPI(title="GIOP Dev Sync Gateway")

GRAPH_URI = os.getenv("GRAPH_DB_URI", "bolt://memgraph-graph:7687")
graph_driver = GraphDatabase.driver(GRAPH_URI, auth=("", ""))
TIMESCALE_URI = os.getenv("TIMESCALE_URI")

class WebhookPayload(BaseModel):
    type: str
    table: str
    record: Optional[Dict[str, Any]] = None
    old_record: Optional[Dict[str, Any]] = None

class TelemetryPayload(BaseModel):
    meter_mrid: str
    active_energy_kwh: float

def sync_to_graph_store(payload: WebhookPayload):
    action = payload.type
    rec = payload.record if action in ["INSERT", "UPDATE"] else payload.old_record
    if not rec or "mrid" not in rec: return

    with graph_driver.session() as session:
        if payload.table == "connectivity_nodes":
            if action in ["INSERT", "UPDATE"]:
                session.run("MERGE (c:ConnectivityNode {mrid: $mrid}) SET c.name = $name", 
                            {"mrid": rec["mrid"], "name": rec.get("name", "Node")})
            elif action == "DELETE":
                session.run("MATCH (c:ConnectivityNode {mrid: $mrid}) DETACH DELETE c", {"mrid": rec["mrid"]})
        elif payload.table == "ac_line_segments":
            if action in ["INSERT", "UPDATE"]:
                session.run("""
                MATCH (s {mrid: $source_id}), (t {mrid: $target_id})
                MERGE (s)-[r:AC_LINE_SEGMENT {mrid: $mrid}]->(t)
                SET r.direction_downstream = $direction, r.phases = $phases, r.voltage = $voltage
                """, {
                    "mrid": rec["mrid"], "source_id": rec["source_node_id"], "target_id": rec["target_node_id"],
                    "direction": rec.get("direction_downstream", True), "phases": rec.get("phases", "ABC"),
                    "voltage": rec.get("nominal_voltage", "MV_11KV")
                })

@app.post("/webhook/supabase-sync")
async def handle_supabase_sync(payload: WebhookPayload, background_tasks: BackgroundTasks):
    background_tasks.add_task(sync_to_graph_store, payload)
    return {"status": "queued"}

@app.post("/api/v1/telemetry/submit")
async def log_meter_interval(payload: TelemetryPayload):
    try:
        conn = psycopg2.connect(TIMESCALE_URI)
        with conn.cursor() as cur:
            cur.execute(
                "INSERT INTO public.meter_readings (meter_mrid, reading_timestamp, active_energy_kwh) VALUES (%s, NOW(), %s);",
                (payload.meter_mrid, payload.active_energy_kwh)
            )
            conn.commit()
        conn.close()
        return {"status": "ingested"}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.get("/api/v1/trace")
async def execute_trace(start_mrid: str):
    query = "MATCH (s {mrid: $mrid}) MATCH p = (s)-[:AC_LINE_SEGMENT*1..10]->(c) RETURN p"
    nodes_out, edges_out = {}, []
    with graph_driver.session() as session:
        res = session.run(query, {"mrid": start_mrid})
        for record in res:
            path = record["p"]
            for n in path.nodes: nodes_out[n["mrid"]] = {"mrid": n["mrid"], "name": n.get("name", "Node"), "type": list(n.labels)}
            for r in path.relationships: edges_out.append({"mrid": r["mrid"], "source": r.start_node["mrid"], "target": r.end_node["mrid"], "phases": r.get("phases", "ABC"), "voltage": r.get("voltage", "MV_11KV")})
    return {"nodes": list(nodes_out.values()), "edges": edges_out}

CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS "postgis";

CREATE TYPE public.asset_lifecycle_state AS ENUM ('PLANNING', 'IN_CONSTRUCTION', 'IN_SERVICE', 'OUT_OF_SERVICE', 'ABANDONED');
CREATE TYPE public.staging_validation_state AS ENUM ('PENDING_FIELD', 'STAGED', 'IN_CONFLICT', 'APPROVED');
CREATE TYPE public.ghana_utility_enum AS ENUM ('ECG_SOUTHERN', 'NEDCO_NORTHERN', 'GRIDCO_TRANSMISSION');
CREATE TYPE public.ghana_voltage_enum AS ENUM ('LV_230V', 'LV_400V', 'MV_11KV', 'MV_33KV', 'HV_161KV', 'HV_330KV');

CREATE TABLE public.identified_objects (
    mrid UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    name VARCHAR(100) NOT NULL,
    lifecycle_state public.asset_lifecycle_state DEFAULT 'PLANNING',
    validation public.staging_validation_state DEFAULT 'PENDING_FIELD',
    error_log TEXT,
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE public.connectivity_nodes (
    mrid UUID PRIMARY KEY REFERENCES public.identified_objects(mrid) ON DELETE CASCADE,
    boundary_feeder_id VARCHAR(50),
    geom GEOMETRY(Point, 4326) NOT NULL
);
CREATE INDEX idx_connectivity_nodes_spatial ON public.connectivity_nodes USING GIST(geom);

CREATE TABLE public.conducting_equipment (
    mrid UUID PRIMARY KEY REFERENCES public.identified_objects(mrid) ON DELETE CASCADE,
    phases VARCHAR(10) DEFAULT 'ABC',
    nominal_voltage public.ghana_voltage_enum NOT NULL,
    serial_number VARCHAR(100) NOT NULL
);

CREATE TABLE public.ac_line_segments (
    mrid UUID PRIMARY KEY REFERENCES public.conducting_equipment(mrid) ON DELETE CASCADE,
    source_node_id UUID REFERENCES public.connectivity_nodes(mrid),
    target_node_id UUID REFERENCES public.connectivity_nodes(mrid),
    direction_downstream BOOLEAN DEFAULT TRUE,
    geom GEOMETRY(LineString, 4326) NOT NULL
);
CREATE INDEX idx_ac_line_segments_spatial ON public.ac_line_segments USING GIST(geom);

CREATE TABLE public.ghana_grid_assets (
    mrid UUID PRIMARY KEY REFERENCES public.identified_objects(mrid) ON DELETE CASCADE,
    operating_utility public.ghana_utility_enum NOT NULL,
    substation_name VARCHAR(100) NOT NULL
);

CREATE OR REPLACE FUNCTION public.intercept_sync_collision()
RETURNS TRIGGER AS $$
BEGIN
    IF EXISTS (SELECT 1 FROM public.identified_objects WHERE mrid = NEW.mrid AND updated_at > OLD.updated_at) THEN
        NEW.validation := 'IN_CONFLICT'::public.staging_validation_state;
        NEW.error_log := 'Conflict Exception: Parallel modification identified.';
    ELSE
        NEW.validation := 'STAGED'::public.staging_validation_state;
        NEW.updated_at := NOW();
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_collision_shield BEFORE UPDATE ON public.identified_objects
FOR EACH ROW WHEN (NEW.validation = 'PENDING_FIELD') EXECUTE FUNCTION public.intercept_sync_collision();
